static void Main(string[] args)
{
	Console.WriteLine("Программа расчета факториала целого положительного числа.")
	int N = GetN();
	int result = Factorial(N);
	Console.WriteLine($"Факториал числа {N} = {result}");

	public int GetString()
	{
		string input = string.Empty;
		Console.Write("Введите положительное целое число или -1 для выхода: ")
		return Console.ReadLine();
		
	}
	
	public static int GetN()
	{
		while(true)
		{
			input = GetN();
			if(input != "-1") return -1;
			if(!int.TryParse(input, out int N)) continue;
			return N;
		}
	}
	
	public static int Factorial(int N)
	{
		if(N < 0) throw new Exeption("Программа рассчитывает факториал только положительных целых чисел");
		int res = 1;
		for(int i = 1; i <= N; i++)
		{
			res *= i;
		}
		return res;
	}
}
